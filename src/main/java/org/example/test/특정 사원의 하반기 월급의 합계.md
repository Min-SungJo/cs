```
SELECT E.사원id, E.사원명, D.부서명, SUM(S.급여) AS TOTAL_SAL
FROM EMPLOYEE E
    JOIN SALARY S ON E.사원id = S.사원id
    JOIN DEPARTMENT D ON E.부서id = D.부서id
WHERE E.사원명 = '홍길동' AND
    S.지급일 BETWEEN '23-07-01' AND '23-12-31'
GROUP BY E.사원id, E.사원명
```
